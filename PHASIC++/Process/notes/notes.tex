\section{CS--FF PS Weight}

The NLO xsec, in CS notation reads
\bea
 \sigma_{NLO}
& = & \sigma_{N}^{(BVI)} + \sigma_{N+1}^{(RS)} \\
& = & \int\limits_N\done\phi\left(\tilde{p}_{ij},\tilde{p}_k\right)
      \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
      \sum\limits_{\tilde{ai},b}
        f_a(\tilde{x}_1,Q^2) f_b(\tilde{x}_2,Q^2)
        \hat\sigma_{ab}^{(BVI)} \nnb\\
&&{}+ \int\limits_{N+1}\done\phi\left(p_i,p_j,p_k\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
      \hat\sigma_{ab}^{(RS)}\,.
\eea
Using identities from CS (5.16) to (5.20) in $D=4$, replacing theta
functions by integration boundaries and inserting the identity place holder $J$,
along with the momenta mappings ($\tilde{p}_{ij}=p_i+p_j-
\frac{\bar{y}_{ijk}}{1-\bar{y}_{ijk}}p_k$, $\tilde{p}_k=
\frac{1}{1-\bar{y}_{ijk}}p_k$, with $\bar{y}_{ijk}=
\frac{p_ip_j}{p_ip_j+p_jp_k+p_kp_i}$ (other $\delta$-functions not necessary to 
define LO kinematics)), into the $N$ parton phase space integral gives
\bea
 \sigma_{NLO}
& = & \int\limits_N\done\phi\left(\tilde{p}_{ij},\tilde{p}_k\right)
      \int\limits_0^1\done z_i\int\limits_0^1\done y_{ijk}
      \int\limits_0^{2\pi}\frac{\done\phi}{2\pi}
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \frac{2\tilde{p}_{ij}\tilde{p}_k}{16\pi^2}(1-y_{ijk})\nnb\\
&&\hspace*{2mm}{}\times
      \left[
            J\;\int\limits_0^1\done\tilde{x}_1\delta(\tilde{x}_1-x_1)
            \int\limits_0^1\done\tilde{x}_2\delta(\tilde{x}_2-x_2)
            \delta(y_{ijk}-\bar{y}_{ijk})
            \sum\limits_{a,b} f_a(\tilde{x}_1,Q^2)
            f_b(\tilde{x}_2,Q^2)
            \hat\sigma_{ab}^{(BVI)}\right. \nnb\\
&&\hspace*{5mm}\left.{}
           +\sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{ab}^{(RS)}\right] \\
& = & \int\limits_N\done\phi\left(p_i,p_j,p_k\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2) \nnb\\
&&\hspace*{2mm}{}\times
      \left[
            J\;\delta(y_{ijk}-\bar{y}_{ijk})
            \hat\sigma_{ab}^{(BVI)}
           +\hat\sigma_{ab}^{(RS)}\right]\,.
\eea
This leaves the momenta mappings in $\hat\sigma_{ab}^{(BVI)}$ implicit
and $J$ to be the solution of the equation
\bea
\lefteqn{\int\limits_N\done\phi\left(\tilde{p}_{ij},\tilde{p}_k\right)
         \int\limits_0^1\done z_i \int\limits_0^1\done y_{ijk}
         \int\limits_0^{2\pi}\frac{\done\phi}{2\pi}
         \int\limits_0^1\done x_1\int\limits_0^1\done x_2
         \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2}\nnb\\
\lefteqn{\hspace*{15mm}\times\;
         \delta(y_{ijk}-\bar{y}_{ijk})
         \delta(\tilde{x}_1-x_1)\delta(\tilde{x}_2-x_2)
         \frac{2\tilde{p}_{ij}\tilde{p}_k}{16\pi^2}(1-y_{ijk})\;J\;
         \mathcal{F}\left(x_1,x_2;\tilde{p}_{ij},\tilde{p}_k\right)}\nnb\\
& = & \int\limits_N\done\phi\left(\tilde{p}_{ij},\tilde{p}_k\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
         \delta(x_1-\tilde{x}_1)\delta(x_2-\tilde{x}_2)
         \frac{2\tilde{p}_{ij}\tilde{p}_k}{16\pi^2}(1-\bar{y}_{ijk})\;J\,
         \mathcal{F}\left(x_1,x_2;\tilde{p}_{ij},\tilde{p}_k\right)\nnb\\
& = & \int\limits_N\done\phi\left(\tilde{p}_{ij},\tilde{p}_k\right)
      \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
         \frac{2\tilde{p}_{ij}\tilde{p}_k}{16\pi^2}(1-\bar{y}_{ijk})\;J\,
         \mathcal{F}\left(\tilde{x}_1,\tilde{x}_2;\tilde{p}_{ij},\tilde{p}_k\right)\nnb\\
& \stackrel{!}{=} & \int\limits_N\done\phi\left(\tilde{p}_{ij},\tilde{p}_k\right)
                    \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
                    \mathcal{F}\left(\tilde{x}_1,\tilde{x}_2;\tilde{p}_{ij},\tilde{p}_k\right)\,.\nnb
\eea
Thus,
\bea
 J^{-1}
& = & \frac{2\tilde{p}_{ij}\tilde{p}_k}{16\pi^2}(1-\bar{y}_{ijk})
\eea
where $\bar{y}_{ijk}$ is the physical relation defined above.





\section{CS--FI PS Weight}

The NLO xsec, in CS notation reads
\bea
 \sigma_{NLO}
& = & \sigma_{N}^{(BVI)} + \sigma_{N+1}^{(RS)} \\
& = & \int\limits_N\done\phi\left(\tilde{p}_{ij};\tilde{p}_a\right)
      \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
      \sum\limits_{a,b}
        f_a(\tilde{x}_1,Q^2) f_b(\tilde{x}_2,Q^2)
        \hat\sigma_{ab}^{(BVI)} \nnb\\
&&{}+ \int\limits_{N+1}\done\phi\left(p_i,p_j;p_a\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
      \hat\sigma_{ab}^{(RS)}\,.
\eea
Using identities from CS (5.45) to (5.48) in $D=4$, replacing theta
functions by integration boundaries and inserting the identity place holder $J$,
along with the momenta mappings ($\tilde{p}_{ij}=p_i+p_j-
(1-\bar{x}_{ij,a})p_a$, $\tilde{p}_a=
\bar{x}_{ij,a}p_a$, with $\bar{x}_{ij,a}=
1-\frac{p_ip_j}{(p_i+p_j)p_a}$ (other $\delta$-function not necessary to 
define LO kinematics)), into the $N$ parton phase space integral gives
\bea
 \sigma_{NLO}
& = & \int\limits_0^1\done x\int\limits_N\done\phi\left(\tilde{p}_{ij};p_a,x\right)
      \int\limits_0^1\done z_i\int\done x_{ij,a}
      \int\limits_0^{2\pi}\frac{\done\phi}{2\pi}
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \frac{2\tilde{p}_{ij}p_a}{16\pi^2}\delta(x-x_{ij,a})\nnb\\
&&\hspace*{2mm}{}\times
      \left[
            J\;\int\limits_0^1\done\tilde{x}_1\delta(\tilde{x}_1-xx_1)
            \int\limits_0^1\done\tilde{x}_2\delta(\tilde{x}_2-x_2)
            \delta(x_{ij,a}-\bar{x}_{ij,a})
            \sum\limits_{a,b} f_a(\tilde{x}_1,Q^2)
            f_b(\tilde{x}_2,Q^2)
            \hat\sigma_{ab}^{(BVI)}\right. \nnb\\
&&\hspace*{5mm}\left.{}
           +\sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{ab}^{(RS)}\right] \\
& = & \int\limits_N\done\phi\left(p_i,p_j;p_a\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2 \nnb\\
&&\hspace*{2mm}{}\times
      \left[
            J\;\delta(x_{ij,a}-\bar{x}_{ij,a})
            \sum\limits_{a,b} f_a(xx_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{ab}^{(BVI)}
           +\sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{ab}^{(RS)}\right]\,.
\eea
This leaves the momenta mappings in $\hat\sigma_{ab}^{(BVI)}$ implicit
and $J$ to be the solution of the equation
\bea
\lefteqn{\int\limits_0^1\done x
         \int\limits_N\done\phi\left(\tilde{p}_{ij};p_a,x\right)
         \int\limits_0^1\done z_i \int\done x_{ij,a}
         \int\limits_0^{2\pi}\frac{\done\phi}{2\pi}
         \int\limits_0^1\done x_1\int\limits_0^1\done x_2
         \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2}\nnb\\
\lefteqn{\hspace*{15mm}\times\;
         \delta(x-x_{ij,a})\delta(x_{ij,a}-\bar{x}_{ij,a})
         \delta(\tilde{x}_1-xx_1)\delta(\tilde{x}_2-x_2)
         \frac{2\tilde{p}_{ij}p_a}{16\pi^2}\;J\;
         \mathcal{F}\left(xx_1,x_2;\tilde{p}_{ij},xp_a\right)}\nnb\\
& = & \int\limits_0^1\done x
      \int\limits_N\done\phi\left(\tilde{p}_{ij};\tilde{p}_a\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
      \frac{1}{x}\delta(x_1-\tfrac{\tilde{x}_1}{x})\delta(x-\bar{x}_{ij,a})
      \frac{2\tilde{p}_{ij}p_a}{16\pi^2}\;J\;
      \mathcal{F}\left(xx_1,x_2;\tilde{p}_{ij},xp_a\right) \nnb\\
& = & \int\limits_N\done\phi\left(\tilde{p}_{ij};\tilde{p}_a\right)
      \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
      \frac{1}{\bar{x}_{ij,a}}
      \frac{2\tilde{p}_{ij}p_a}{16\pi^2}\;J\;
      \mathcal{F}\left(\tilde{x}_1,\tilde{x}_2;\tilde{p}_{ij},\tilde{p}_a\right) \nnb\\
& \stackrel{!}{=} & \int\limits_N\done\phi\left(\tilde{p}_{ij},\tilde{p}_k\right)
                    \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
                    \mathcal{F}\left(\tilde{x}_1,\tilde{x}_2;\tilde{p}_{ij},\tilde{p}_k\right)\,.\nnb
\eea
Thus,
\bea
 J^{-1}
& = & \frac{2\tilde{p}_{ij}p_a}{16\pi^2}\frac{1}{\bar{x}_{ij,a}}
\eea
where $\bar{x}_{ij,a}$ is the physical momentum fraction of $\tilde{p}_{ai}$ of
$p_a$.

\eea
where $\bar{x}_{i,ab}$ is the physical momentum fraction of $\tilde{p}_{ai}$ of

\eea

\bea
 J^{-1}
& = & \frac{2p_ap_b}{16\pi^2}\frac{1-\bar{x}_{i,ab}}{\bar{x}_{i,ab}}
\eea
where $\bar{x}_{i,ab}$ is the physical momentum fraction of $\tilde{p}_{ai}$ of
$p_a$.


\section{Dependence of the Result on the Choice of the form of the $\delta$-Function involving $x_1$ and $\tilde{x_1}$}

Consider the CS--II PS Weight case above. When performing the calculations using
$\delta(x_1-\frac{\tilde{x}_1}{x}$ instead of $\delta(\tilde{x}_1-xx_1)$ the 
result reads
\bea
 \sigma_{NLO}
& = & \int\limits_0^1\done x\int\limits_N\done\phi\left(\tilde{k}_1,\ldots,\tilde{k}_N;xp_a+p_b\right)
      \int\limits_0^{1-x}\done\tilde{v}_i\int\done x_{i,ab}
      \int\limits_0^{2\pi}\frac{\done\phi}{2\pi}
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \frac{2p_ap_b}{16\pi^2}\delta(x-x_{i,ab})\nnb\\
&&\hspace*{2mm}{}\times
      \left[
            J\;\int\limits_0^1\done\tilde{x}_1\delta(x_1-\tfrac{\tilde{x}_1}{x})
            \int\limits_0^1\done\tilde{x}_2\delta(x_2-\tilde{x}_2)
            \delta(x_{i,ab}-\bar{x}_{i,ab})
            \sum\limits_{\tilde{ai},b} f_{\tilde{ai}}(\tilde{x}_1,Q^2)
            f_b(\tilde{x}_2,Q^2)
            \hat\sigma_{\tilde{ai}b}^{(BVI)}\right. \nnb\\
&&\hspace*{5mm}\left.{}
           +\sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{ab}^{(RS)}\right] \\
& = & \int\limits_0^1\done x\int\limits_N\done\phi\left(\tilde{k}_1,\ldots,\tilde{k}_N;xp_a+p_b\right)
      \int\limits_0^{1-x}\done\tilde{v}_i\int\done x_{i,ab}
      \int\limits_0^{2\pi}\frac{\done\phi}{2\pi}
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \frac{2p_ap_b}{16\pi^2}\delta(x-x_{i,ab})\nnb\\
&&\hspace*{2mm}{}\times
      \left[
            Jx\;\int\limits_0^1\done\tilde{x}_1\delta(\tilde{x}_1-xx_1)
            \int\limits_0^1\done\tilde{x}_2\delta(x_2-\tilde{x}_2)
            \delta(x_{i,ab}-\bar{x}_{i,ab})
            \sum\limits_{\tilde{ai},b} f_{\tilde{ai}}(\tilde{x}_1,Q^2)
            f_b(\tilde{x}_2,Q^2)
            \hat\sigma_{\tilde{ai}b}^{(BVI)}\right. \nnb\\
&&\hspace*{5mm}\left.{}
           +\sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{ab}^{(RS)}\right] \\
& = & \int\limits_N\done\phi\left(p_i,k_1,\ldots,k_N;p_a+p_b\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2 \nnb\\
&&\hspace*{2mm}{}\times
      \left[
            J\bar{x}_{i,ab}\;\delta(x_{i,ab}-\bar{x}_{i,ab})
            \sum\limits_{\tilde{ai},b} f_{\tilde{ai}}(xx_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{\tilde{ai}b}^{(BVI)}
           +\sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{ab}^{(RS)}\right]\nnb\\
& = & \int\limits_N\done\phi\left(p_i,k_1,\ldots,k_N;p_a+p_b\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2 \nnb\\
&&\hspace*{2mm}{}\times
      \left[
            \bar{J}\;\delta(x_{i,ab}-\bar{x}_{i,ab})
            \sum\limits_{\tilde{ai},b} f_{\tilde{ai}}(xx_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{\tilde{ai}b}^{(BVI)}
           +\sum\limits_{a,b} f_a(x_1,Q^2) f_b(x_2,Q^2)
            \hat\sigma_{ab}^{(RS)}\right]\,.
\eea
Note the extra factor $\bar{x}_{i,ab}$ occuring above being absorbed in $bar{J}$.
This then is determined as
\bea
\lefteqn{\int\limits_0^1\done x\int\limits_N\done\phi\left(\tilde{k}_1,\ldots,\tilde{k}_N;xp_a+p_b\right)
         \int\limits_0^{1-x}\done\tilde{v}_i\int\done x_{i,ab}
         \int\limits_0^{2\pi}\frac{\done\phi}{2\pi}
         \int\limits_0^1\done x_1\int\limits_0^1\done x_2
         \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2}\nnb\\
\lefteqn{\hspace*{15mm}\times\;
         \delta(x_{i,ab}-\bar{x}_{i,ab})
         \delta(x-x_{i,ab})\delta(x_1-\tfrac{\tilde{x}_1}{x})\delta(x_2-\tilde{x}_2)
         \frac{2p_ap_b}{16\pi^2}\;J\;
         \mathcal{F}\left(xx_1,x_2;xp_a,p_b,\tilde{k}_j\right)}\nnb\\
& = & \int\limits_0^1\done x\int\limits_N\done\phi\left(\tilde{k}_1,\ldots,\tilde{k}_N;xp_a+p_b\right)
      \int\limits_0^1\done x_1\int\limits_0^1\done x_2
      \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2\nnb\\
\lefteqn{\hspace*{25mm}\times\;
         \delta(x-\bar{x}_{i,ab})
         \delta(x_1-\tfrac{\tilde{x}_1}{x})\delta(x_2-\tilde{x}_2)
         \frac{2p_ap_b}{16\pi^2}(1-x)\;J\;
         \mathcal{F}\left(xx_1,x_2;xp_a,p_b,\tilde{k}_j\right)}\nnb\\
& = & \int\limits_N\done\phi\left(\tilde{k}_1,\ldots,\tilde{k}_N;\tilde{p}_{ai}+p_b\right)
      \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
      (1-\bar{x}_{i,ab})\frac{2p_ap_b}{16\pi^2}\;J\;
      \mathcal{F}\left(\tilde{x}_1,\tilde{x}_2;\tilde{p}_{ai},p_b,\tilde{k}_j\right)\nnb\\
& \stackrel{!}{=} & \int\limits_N\done\phi\left(\tilde{k}_1,\ldots,\tilde{k}_N;\tilde{p}_{ai}+p_b\right)
                    \int\limits_0^1\done\tilde{x}_1\int\limits_0^1\done\tilde{x}_2
                    \mathcal{F}\left(\tilde{x}_1,\tilde{x}_2;\tilde{p}_{ai},p_b,\tilde{k}_j\right)\,.\nnb
\eea
Thus, the needed quantity $\bar{J}=J\bar{x}_{i,ab}$ reads
\bea
 \bar{J}^{-1}
& = & J^{-1} \bar{x}_{i,ab}^{-1} \nnb\\
& = & \frac{2p_ap_b}{16\pi^2}\frac{1-\bar{x}_{i,ab}}{\bar{x}_{i,ab}}
\eea
where $\bar{x}_{i,ab}$ is the physical momentum fraction of $\tilde{p}_{ai}$ of
$p_a$.
